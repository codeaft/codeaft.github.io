codeaft@codeaft:~$ octave
...
octave> codeaft = "Koding"
codeaft = Koding

octave> codeaft = [codeaft, "window"]
codeaft = Codeaft

octave> typeinfo(codeaft)
ans = string

octave> codeaft(6)
ans = g

octave> codeaft(2:11)
ans = odingwindo

octave> length(codeaft)
ans = 12

octave> fliplr(codeaft)
ans = wodniwgnidoK

octave> toupper(codeaft)
ans = CODEAFT

octave> tolower(codeaft)
ans = codeaft

octave> sort(codeaft)
ans = Kddgiinnooww

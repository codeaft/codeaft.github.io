class Codeaft
{
    int a;
    Codeaft()
    {
        this(30);
    }
    Codeaft(int a)
    {
        this.a=a;
        System.out.println(a);
    }
    public static void main(String args[])
    {
        Codeaft c1=new Codeaft();
    }
}

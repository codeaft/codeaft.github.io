class Codeaft
{
    public static int add(int a,int b)
    {
        a=100;
        b=100;
        int c=a+b;
        return c;
    }
    public static void main(String args[])
    {
        int a=20,b=10,c;
        c=a+b;
        add(a,b);
        System.out.println(a+b);
    }
}

class Codeaft
{
    int a=100,b=100;
    public void add(Codeaft c)
    {
        c.a=c.a+c.b+500;
    }
    public static void main(String args[])
    {
        Codeaft c=new Codeaft();
        c.a=c.a+c.b;
        System.out.println(c.a);
        c.add(c);
        System.out.println(c.a);
    }
}
{%- endhighlight -%}

{{ page.title }}
{{ site.javafile }}
{%- highlight java -%}
class Codeaft
{
    public static void main()
    {
        System.out.println("Welcome to Codeaft");
    }
    public static void main(String args[])
    {
        System.out.println("Hello, World!");
        main();
    }
}
{%- endhighlight -%}
{{ site.output }}
{{ site.runjava }}
Hello, World!
Welcome to Codeaft
codeaft@codeaft:~$

class Codeaft
{
    public static void main(String args) 
    {
        System.out.println("Welcome to "+args);
    }
    public static void main(String[] args) 
    {
        System.out.println("Hello, World!");
        main("Codeaft");
    }
}
{%- endhighlight -%}
{{ site.output }}
{{ site.runjava }}
Hello, World!

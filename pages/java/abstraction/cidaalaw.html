abstract class Arithmetic
{
    int add(int a,int b)
    {
        int c=a+b;
        return c;
    }
    void display(int c)
    {
        System.out.println("Addition "+c);
    }
}

class Codeaft extends Arithmetic
{
    public static void main(String[] args) 
    {
        Codeaft c=new Codeaft();
        int addition=c.add(10,10);
        c.display(addition);
    }
}

abstract class Fruit
{
    abstract void cutting();
}

class Mango extends Fruit
{
    void cutting()
    {
        System.out.println("Cutting Mango");
    }
}

class Apple extends Fruit
{
    void cutting()
    {
        System.out.println("Cutting Apple");
    }
}

class Codeaft
{
    public static void main(String[] args) 
    {
        Mango m=new Mango();
        m.cutting();
        Apple a=new Apple();
        a.cutting();
    }
}

{{ site.javafile }}
{%- highlight java -%}
abstract class Shape
{
    abstract void display();
    abstract void area(float b,float h);
}

class Triangle extends Shape
{
    void display() 
    {
        System.out.print("Area of Triangle ");
    }
    void area(float b, float h) 
    {
        System.out.println((b*h)/2);
    }
}

class Rectangle extends Shape
{
    void display() 
    {
        System.out.print("Area of Rectangle ");
    }
    void area(float b, float h) 
    {
        System.out.println(b*h);
    }
}

class Codeaft
{
    Codeaft(float a)
    {
        System.out.println("Area of Square "+(a*a));
    }

    public static void main(String[] args) 
    {
        Triangle t=new Triangle();
        t.display();
        t.area(10,20);
        Rectangle r=new Rectangle();
        r.display();
        r.area(10,20);
        Codeaft c=new Codeaft(10);
    }
}
{%- endhighlight -%}

{%- highlight java -%}
abstract class Server
{
    String name;
    Server(String name)
    { 
        this.name = name; 
    }
    public abstract void start();
}

class Mongodb extends Server
{
    Mongodb(String name)
    {
        super(name);
    }
    public void start() 
    { 
        System.out.println( this.name + " Started Successfully"); 
    }
}

class Codeaft
{
    public static void main(String[] args) 
    {
        Server s=new Mongodb("MongoDB Server");
        s.start();
    }
}

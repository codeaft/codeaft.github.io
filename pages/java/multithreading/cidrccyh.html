class Codeaft implements Runnable
{  
    public void run()
    {  
        System.out.println("Thread Running...");  
    }  
    public static void main(String args[])
    {  
        Codeaft c=new Codeaft();
        Thread t1=new Thread(c); 
        //t1.run();  
        System.out.println(t1.isAlive()); 
        t1.start(); 
        System.out.println(t1.isAlive());
    }  
}

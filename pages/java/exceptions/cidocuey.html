class Codeaft
{
    public static void main(String args[])
    {
        String s=null;
        try
        {
            System.out.println("Length: "+s.length());
        }
        catch(NullPointerException e)
        {
            System.out.println("String is null, hence, Unable to find the length.");
        }
    }
}

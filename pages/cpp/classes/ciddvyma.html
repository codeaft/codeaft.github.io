#include 
using namespace std;
class codeaft
{
public:
    codeaft()
    {
        cout<<"\nConstructor is called";
    }
    ~codeaft()
    {
        cout<<"\nDestructor is called";
    }
};
int main()
{
    codeaft c;
    codeaft();
    c.~codeaft();
}
